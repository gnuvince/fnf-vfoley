\begin{frame}
  \centering{\Huge ed}
  \vskip.5cm
  \overview{Rich addressing}{
    \begin{itemize}
      \item Ken Thompson, c.~1971
      \item original UNIX editor
      \item line-oriented
      \item used on teletypes
      \item minimal output
      \item \emph{rich addressing syntax}
      \item surprising power that
        many modern editors lack
    \end{itemize}
  }{
    \begin{itemize}
      \item \texttt{.} (\emph{dot}): current line
      \item $n$ : line number $n$
      \item $m,n$: line range
      \item \texttt{\$}: last line
      \item \texttt{\%}: equivalent to \texttt{1,\$}
      \item \texttt{/re/}: find \texttt{re} forward
      \item \texttt{?re?}: find \texttt{re} backward
      \item \texttt{%
        /\^{}main/;%
        /\^{}$\}$/
      }: \emph{main} function
      \item \texttt{g/re/p}: \emph{grep}
    \end{itemize}
  }
\end{frame}

\begin{frame}
  \centering{\Huge sam}
  \vskip.5cm
  \overview{Remote file editing}{
    \begin{itemize}
      \item Rob Pike, c.~1987
      \item evolution of ed
      \item visual editor
      \item more powerful addressing
      \item multi-file support
      \item \emph{remote file editing}
    \end{itemize}
  }{
    \begin{itemize}
      \item remote host: command interpreter;
        local host: GUI
      \item must be installed on both end-points
      \item small chunks, not whole file
    \end{itemize}
  }
\end{frame}

\begin{frame}
  \centering{\Huge acme}\\
  \centering{(or: How I Learned to Stop Worrying And Love Unix)}
  \vskip.5cm

  \overview{Unix integration}{
    \begin{itemize}
      \item Rob Pike, c.~1994
      \item full-screen editor
      \item mouse-oriented
      \item few built-in commands
      \item \emph{strong Unix integration}
      \item \textsl{A Tour of Acme}: {\footnotesize \url{https://research.swtch.com/acme}}
    \end{itemize}
  }{
    \begin{itemize}
      \item \texttt{cmd}: output in new buffer
      \item \texttt{<cmd}: output in current buffer
      \item \texttt{>cmd}: selection as input, output in new buffer
      \item \texttt{|cmd}: selection as input, output in current buffer
    \end{itemize}
  }
\end{frame}


\begin{frame}
  \centering{\Huge vi}
  \vskip.5cm

  \overview{Modal editing}{
    \begin{itemize}
      \item Bill Joy, 1976
      \item \textit{vi}sual interface for ex
      \item most popular Unix editor
      \item worth learning since it's so widely available
      \item \emph{modal editing}
    \end{itemize}
  }{
    \begin{itemize}
      \item normal mode: edit text
      \item insert mode: input text
      \item more commands, \\less RSI-inducing
      \item combinable commands: \\edit + motion (\texttt{d\$}, \texttt{c3w})
      \item some combined commands don't behave ``like they should''
    \end{itemize}
  }
\end{frame}

\begin{frame}
  \centering{\Huge Emacs}
  \vskip.5cm

  \overview{Keyboard macros}{
    \begin{itemize}
      \item Richard Stallman, 1976 \& 1985
      \item 2nd most popular Unix editor
      \item $\sim10\%$ C, $\sim90\%$ Lisp
      \item extremely customizable
      \item \emph{keyboard macros}
    \end{itemize}
  }{
    \begin{itemize}
      \item Record keystrokes
      \item Replay them
      \item Useful when regexes not easy or possible
      \item Careful with invariants!
    \end{itemize}
  }
\end{frame}
